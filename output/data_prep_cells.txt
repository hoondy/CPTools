from scipy.sparse import csr_matrix

def cp2h5ad(prefix, CellType, Batch, res_path, meta_path):

    # read res
    res = pd.read_csv(res_path, sep='\t', skiprows=8)

    # CellType
    res['CellType'] = CellType
    
    # Batch_row_col
    res['Batch_row_col'] = ['_'.join([Batch, str(x), str(y)]) for x, y in zip(res.Row, res.Column)]
    del res['Row']
    del res['Column']
    
    # drop columns with NaN
    res = res.dropna(axis=1, how='all')

    # read meta
    meta = pd.read_csv(meta_path)

    # merged
    res = res.set_index("Batch_row_col")
    meta = meta.set_index("Batch_row_col")
    merged = pd.merge(res, meta, how='left', on='Batch_row_col')
    merged = merged[~merged.Batch.isna()].copy()
    
    # mat
    dat = merged.loc[:,[x.startswith('Nuclei Selected - ') for x in merged.columns]].copy()
    mat = csr_matrix(dat, dtype=np.float32)

    # obs
    obs = merged.loc[:,[not x.startswith('Nuclei Selected - ') for x in merged.columns]].copy()
    obs.index = ['_'.join([str(x), str(y)]) for x, y in zip(obs.Batch, obs.Well)]

    # var
    var = pd.DataFrame({'feature':dat.columns})
    var['name'] = [x.replace('Nuclei Selected - ','').replace(' - Mean per Well','').replace('PhenoVue Hoechst 33342','Nuclei').replace('PhenoVue Fluor 488','ER').replace('PhenoVue Fluor 555','AGP_NR').replace('PhenoVue 641 Mito Stain','Mito') for x in var.feature]
    var.index = var['name']

    # adata
    adata = ad.AnnData(mat)
    adata.obs = obs
    adata.var = var
    
    # raw
    adata.layers["raw"] = adata.X.copy()

    # additional annotations: DMSO or Drug
    adata.obs['DMSO'] = ['DMSO' if x == 'DMSO' else 'Drug' for x in adata.obs['Treatment']]
    adata.obs.DMSO = adata.obs.DMSO.astype('category')
    
    # additional annotations: MOA
    adata.obs['MOA'] = adata.obs.PathWay.astype('object')
    adata.obs.loc[adata.obs.MOA.isna(),'MOA'] = 'DMSO'
    adata.obs['MOA'] = [str(x).split(';')[0]for x in adata.obs.MOA]
    adata.obs.MOA = adata.obs.MOA.astype('category')
    
    # additional annotations: PathWay
    adata.obs.PathWay = adata.obs.PathWay.astype('object')
    adata.obs.loc[adata.obs.PathWay.isna(),'PathWay'] = 'DMSO'
    adata.obs.PathWay = adata.obs.PathWay.astype('category')
    
    # additional annotations: Research_Area
    adata.obs.Research_Area = adata.obs.Research_Area.astype('object')
    adata.obs.loc[adata.obs.Research_Area.isna(),'Research_Area'] = 'DMSO'
    adata.obs.Research_Area = adata.obs.Research_Area.astype('category')

    # save
    adata.write(prefix+'_'+CellType+'_'+Batch+'.h5ad')
