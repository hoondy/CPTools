import numpy as np
import pandas as pd
from scipy.stats import median_abs_deviation
import warnings

def robust_zscore_normalize(adata, batch_key='Batch', treatment_key='Treatment', control_val='DMSO', epsilon=1e-6):
    """
    Performs robust Z-score normalization per batch using control samples.
    
    Formula: (X - Median_Control) / (MAD_Control * 1.4826 + epsilon)
    
    Parameters:
    - adata: AnnData object
    - batch_key: Column in adata.obs distinguishing plates
    - treatment_key: Column in adata.obs distinguishing treatments
    - control_val: Value in treatment_key representing the control (e.g., 'DMSO')
    - epsilon: Small value to prevent division by zero
    """
    
    # 1. Create a container for the normalized data (copy of shape)
    # We use a float32 array to save memory
    normalized_X = np.zeros_like(adata.X.todense(), dtype=np.float32)
    
    # 2. Get unique batches (plates)
    batches = adata.obs[batch_key].unique()
    print(f"Processing {len(batches)} batches...")

    for batch in batches:
        # Indices for the current batch
        batch_mask = adata.obs[batch_key] == batch
        
        # Indices for controls within this batch
        control_mask = batch_mask & (adata.obs[treatment_key] == control_val)
        
        # Check if controls exist in this batch
        if not np.any(control_mask):
            warnings.warn(f"Batch {batch} has no '{control_val}' controls. Skipping normalization for this batch.")
            continue
            
        # 3. Extract Control Data for this Batch
        # We assume adata.X is a numpy array. If sparse, use .toarray()
        control_data = adata.X.todense()[control_mask, :]
        batch_data = adata.X.todense()[batch_mask, :]
        
        # 4. Calculate Median and MAD (scaled to consistent sigma)
        # scale='normal' automatically applies the 1.4826 scaling factor
        control_median = np.median(control_data, axis=0)
        control_mad = median_abs_deviation(control_data, axis=0, scale='normal')
        
        # 5. Apply Transformation: (X - Median) / (MAD + epsilon)
        # We add epsilon to handle features that are constant (MAD=0)
        transformed_batch = (batch_data - control_median) / (control_mad + epsilon)
        
        # 6. Store back in the main array
        normalized_X[batch_mask, :] = transformed_batch

    # 7. Save to AnnData
    # Best practice: Keep raw data in .X or a layer, save normalized to a new layer
    adata.layers['raw'] = adata.X.copy() # Backup raw data
    adata.X = normalized_X # Update .X with normalized data
    
    print("Robust Z-score normalization complete.")
    return adata

def unsupervised_feature_reduction(adata, variance_threshold=1e-2):
    """
    Performs unsupervised feature selection on an AnnData object.
    
    Steps:
    1. Blocklist (Location/Execution features) see https://github.com/cytomining/pycytominer/blob/main/pycytominer/data/blocklist_features.txt
    2. Drop NaNs
    3. Low Variance Filter
    4. High Correlation Filter
    """
    original_n_vars = adata.n_vars
    print(f"Starting Feature Reduction. Initial features: {original_n_vars}")
    
    # --- STEP 1: Blocklist Removal ---
    # In Cell Painting, 'Location' features (Center_X, Center_Y) are often technical artifacts 
    # capturing where the cell is in the image, not its biology.
    # 'Execution' or 'Euler' features typically track segmentation quality, not biology.
    
    blocklist_keywords = ['Manders', 'RWC', 'Location', 'Granularity', 'Execution', 'Euler']
    
    # Identify features to keep (those NOT containing blocklist keywords)
    keep_mask = ~adata.var_names.str.contains('|'.join(blocklist_keywords))
    
    # Filter the AnnData object
    adata = adata[:, keep_mask].copy()
    print(f"Step 1 (Blocklist): Dropped {original_n_vars - adata.n_vars} features.")
    
    # --- STEP 2: Drop features with NaNs ---
    # Drop columns that have ANY NaNs.
    
    # Check for NaNs in columns (features)
    # If using sparse matrix, convert to dense first: X = adata.X.toarray()
    X = adata.X 
    nan_mask = np.isnan(X).any(axis=0)
    adata = adata[:, ~nan_mask].copy()
    print(f"Step 2 (NaNs): Dropped {np.sum(nan_mask)} features containing NaNs.")

    # --- STEP 3: Low Variance Filter ---
    # Remove features that don't change (variance near 0).
    # We use sklearn's VarianceThreshold for speed.
    
    selector = VarianceThreshold(threshold=variance_threshold)
    try:
        selector.fit(adata.X)
        adata = adata[:, selector.get_support()].copy()
        print(f"Step 3 (Variance): Dropped features with variance < {variance_threshold}. Remaining: {adata.n_vars}")
    except ValueError:
        print("Warning: Variance Threshold failed (maybe all features are low variance?). Skipping.")

    print(f"Final Feature Count: {adata.n_vars} ({(adata.n_vars/original_n_vars)*100:.1f}% of original)")
    
    return adata

import pandas as pd
import numpy as np

def variance_based_correlation_reduction(adata, threshold=0.95):
    """
    Removes redundant features by clustering highly correlated ones 
    and keeping ONLY the feature with the highest variance per cluster.
    
    Strategy:
    1. Calculate correlation matrix.
    2. Sort features by Variance (High to Low).
    3. Iterate through sorted features:
       - Keep the current feature (it has the highest variance of its group).
       - Drop all other features that are highly correlated with it.
    """
    print(f"Starting Correlation Reduction on {adata.n_vars} features...")
    
    # 1. Calculate Correlation Matrix
    # Use float32 to save memory
    df = pd.DataFrame(adata.X, columns=adata.var_names, dtype=np.float32)
    corr_matrix = df.corr().abs()
    
    # 2. Calculate Variances
    variances = df.var()
    
    # 3. Sort features by Variance (High -> Low)
    # This is the "magic" step. By processing high-variance features first,
    # we ensure they become the "representative" of their correlation cluster.
    sorted_features = variances.sort_values(ascending=False).index.tolist()
    
    features_to_drop = set()
    features_to_keep = []
    
    # 4. Iterate
    for feature in sorted_features:
        if feature in features_to_drop:
            continue
            
        # This feature is the "winner" (highest variance among remaining).
        features_to_keep.append(feature)
        
        # Find all features that are highly correlated with this winner
        # (row in correlation matrix > threshold)
        correlated_mask = corr_matrix[feature] > threshold
        
        # Identify features to drop (excluding the winner itself)
        correlated_features = corr_matrix.index[correlated_mask].tolist()
        
        # Add them to the drop set
        # We don't need to check variances anymore because we sorted by variance!
        # Since 'feature' is the first one we met in the sorted list, it IS the highest variance.
        for f in correlated_features:
            if f != feature:
                features_to_drop.add(f)
                
    # 5. Apply Drop
    # Re-order to match original AnnData order for consistency
    features_to_keep_set = set(features_to_keep)
    final_features = [f for f in adata.var_names if f in features_to_keep_set]
    adata_reduced = adata[:, final_features].copy()
    
    print(f"Reduction: Dropped {len(features_to_drop)} redundant features.")
    print(f"Kept {len(final_features)} features.")
    
    return adata_reduced

def feature_selection_by_replicates(adata, treatment_key='Treatment', control_key='DMSO', quantile_threshold=0.2):
    """
    Selects features based on reproducibility (Signal-to-Noise Ratio).
    
    Parameters:
    - adata: AnnData object (Normalized & Batch Corrected)
    - treatment_key: Column name for drug/treatment labels
    - control_key: Value for control samples (usually 'DMSO')
    - quantile_threshold: Fraction of features to DROP (e.g., 0.2 drops bottom 20%)
    
    Returns:
    - adata: Filtered AnnData object
    """
    print(f"Starting Replicate Signal Analysis on {adata.n_vars} features...")
    
    # 1. Prepare Data
    # We work with a DataFrame for easier grouping
    # Using float32 to save memory
    df = pd.DataFrame(adata.X, columns=adata.var_names, index=adata.obs_names)
    df[treatment_key] = adata.obs[treatment_key].values
    
    # Exclude Control (DMSO) from the 'Signal' calculation
    treatments = df[df[treatment_key] != control_key]
    
    # 2. Calculate Signal (Variance Between Treatment Means)
    # Group by Drug -> Calculate Mean Vector for each Drug -> Calculate Variance of those Means
    treatment_means = treatments.groupby(treatment_key).mean()
    signal_variance = treatment_means.var(axis=0) # Vector of shape (n_features,)
    
    # 3. Calculate Noise (Average Variance Within Replicates)
    # Group by Drug -> Calculate Variance Vector for each Drug -> Average these Variances
    # We use ALL data (including DMSO) to estimate the typical technical noise
    within_group_variance = df.groupby(treatment_key).var()
    noise_variance = within_group_variance.mean(axis=0) # Vector of shape (n_features,)
    
    # 4. Calculate SNR
    # Add small epsilon to avoid division by zero
    snr_ratios = signal_variance / (noise_variance + 1e-9)
    
    # 5. Determine Cutoff
    # We drop the features with the lowest SNR (bottom X%)
    cutoff_value = snr_ratios.quantile(quantile_threshold)
    
    # Identify features to keep
    features_to_keep_mask = snr_ratios > cutoff_value
    features_to_keep = snr_ratios.index[features_to_keep_mask].tolist()
    
    # 6. Apply Filter to AnnData
    # Sort to maintain original order
    final_features = [f for f in adata.var_names if f in features_to_keep]
    adata_reduced = adata[:, final_features].copy()
    
    # --- Reporting ---
    print(f"SNR Threshold: {cutoff_value:.4f}")
    print(f"Dropped {adata.n_vars - len(final_features)} noisy features (Bottom {quantile_threshold*100}%).")
    print(f"Remaining Features: {len(final_features)}")
    
    # Optional: Save the SNR scores in var for future reference
    adata_reduced.var['replicate_snr'] = snr_ratios[final_features]
    
    return adata_reduced

import numpy as np
import pandas as pd
from scipy import linalg

def zca_whitening(adata, treatment_key='Treatment', control_key='DMSO', epsilon=1e-5):
    """
    Performs Regularized ZCA Whitening on an AnnData object.
    
    Logic:
    1. Compute Covariance Matrix of Control (DMSO) samples only.
    2. Perform SVD (Singular Value Decomposition) on Covariance.
    3. Compute Whitening Matrix: W = U * diag(1 / sqrt(eigenvalues + epsilon)) * U.T
    4. Transform all data: X_whitened = X dot W
    
    Parameters:
    - adata: AnnData object (Already normalized & feature selected)
    - epsilon: Regularization parameter (prevents division by zero for low-variance directions)
    """
    print("Starting ZCA Whitening transformation...")
    
    # 1. Extract Data
    # Ensure data is centered (Robust Z-score should have done this, but we enforce mean=0 for controls)
    X = adata.X
    
    # Identify Controls
    control_mask = (adata.obs[treatment_key] == control_key).values
    X_controls = X[control_mask, :]
    
    # Center controls strictly to 0 (Critical for covariance calculation)
    # We subtract the control mean from EVERYTHING (Controls & Treatments)
    control_mean = X_controls.mean(axis=0)
    X_centered = X - control_mean
    X_controls_centered = X_controls - control_mean
    
    # 2. Compute Covariance Matrix (Sigma) of CONTROLS
    # Shape: (n_features, n_features)
    # Note: If n_features > n_controls, this is rank deficient (hence epsilon)
    n_controls = X_controls_centered.shape[0]
    sigma = np.dot(X_controls_centered.T, X_controls_centered) / (n_controls - 1)
    
    # 3. Eigen Decomposition (SVD is more robust numerically)
    # U: Eigenvectors, S: Eigenvalues
    try:
        U, S, _ = linalg.svd(sigma)
    except linalg.LinAlgError:
        print("SVD failed to converge. Trying standard eigendecomposition...")
        S, U = linalg.eigh(sigma)
        # Sort eigenvalues/vectors in descending order
        idx = S.argsort()[::-1]
        S = S[idx]
        U = U[:, idx]
        
    # 4. Build Whitening Matrix
    # Formula: W = U * diag(1/sqrt(S + epsilon)) * U.T
    # Regularization (epsilon) is crucial for Cell Painting to handle noise
    
    # Clip negative eigenvalues (numerical errors) to 0
    S = np.clip(S, 0, None)
    
    # Calculate inverse square root of eigenvalues
    s_inv_sqrt = np.diag(1.0 / np.sqrt(S + epsilon))
    
    # ZCA Matrix
    W = np.dot(np.dot(U, s_inv_sqrt), U.T)
    
    # 5. Apply Transformation to ALL data
    X_whitened = np.dot(X_centered, W)
    
    # 6. Store in AnnData
    # We save the whitening result to a new layer or overwrite X
    adata.layers['pre_whitening'] = adata.X.copy()
    adata.X = X_whitened
    
    print("ZCA Whitening Complete.")
    
    return adata

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from scipy import stats
from adjustText import adjust_text  # pip install adjustText
from matplotlib.legend_handler import HandlerTuple

def visualize_drug_mechanism(adata, drug_code, control_code, layer='raw', top_n=5):
    """
    Generates a Volcano Plot and Boxplots for a specific Drug vs. Control comparison.
    
    Parameters:
    - adata: AnnData object
    - drug_code: Value in 'PlateSet_DrugCode' for treatment (e.g., '13A_Drug88')
    - control_code: Value in 'PlateSet_DrugCode' for control (e.g., '13A_DMSO')
    - layer: Layer containing INTERPRETABLE data (e.g., 'normalized' or 'raw')
    """
    print(f"Comparing {drug_code} vs. {control_code} using layer '{layer}'...")
    
    # 1. Extract Data
    # We use the specified layer to ensure biological interpretability
    try:
        data_matrix = adata.layers[layer]
    except KeyError:
        print(f"Layer '{layer}' not found! Using .X (Warning: Ensure .X is not whitened!)")
        data_matrix = adata.X

    # Create a temporary DataFrame for easier slicing
    df = pd.DataFrame(data_matrix, columns=adata.var_names, index=adata.obs_names)
    df['Condition'] = adata.obs['PlateSet_DrugCode'].values
    
    # Subset to the two conditions
    drug_df = df[df['Condition'] == drug_code].drop(columns='Condition')
    ctrl_df = df[df['Condition'] == control_code].drop(columns='Condition')
    
    if len(drug_df) < 2 or len(ctrl_df) < 2:
        print("Error: Not enough replicates to perform stats.")
        return

    # 2. Calculate Statistics (Vectorized)
    # Effect Size: Difference of Means (Robust Z-score units)
    # Since data is log-normalized or Z-scored, Diff = Log2 Fold Change equivalent
    effect_size = drug_df.mean(axis=0) - ctrl_df.mean(axis=0)
    
    # Significance: T-test (Indep)
    # We use Welch's t-test (equal_var=False) because variance might differ
    t_stat, p_vals = stats.ttest_ind(drug_df, ctrl_df, axis=0, equal_var=False)
    
    # Handle NaNs (if variance is 0)
    p_vals = np.nan_to_num(p_vals, nan=1.0)
    
    # Log transform P-values for plotting
    # Add small epsilon to avoid log(0)
    nlog10_p = -np.log10(p_vals + 1e-300)
    
    # Create Results DataFrame
    results = pd.DataFrame({
        'Feature': adata.var_names,
        'Effect_Size': effect_size,
        'Log_P': nlog10_p,
        'P_Val': p_vals
    }).set_index('Feature')
    
    # 3. Identify Top Hits
    # Criteria: High Effect AND High Significance
    # We sort by a combined score or just P-value
    top_hits = results.sort_values('Log_P', ascending=False).head(top_n)
    
    # --- PLOT 1: VOLCANO PLOT ---
    plt.figure(figsize=(10, 6))
    
    # Plot all points (grey)
    plt.scatter(results['Effect_Size'], results['Log_P'], c='lightgrey', alpha=0.5, s=10)
    
    # Highlight significant hits (Red)
    # Thresholds: P < 0.05 (LogP > 5) and Effect > 1.0 (1 sigma shift)
    sig_mask = (results['P_Val'] < 0.01) & (results['Effect_Size'].abs() >= 1.0)
    plt.scatter(results.loc[sig_mask, 'Effect_Size'], results.loc[sig_mask, 'Log_P'], c='firebrick', s=20, alpha=0.8)
    
    # Label top N features
    texts = []
    for feature in top_hits.index:
        x, y = results.loc[feature, 'Effect_Size'], results.loc[feature, 'Log_P']
        plt.scatter(x, y, c='red', edgecolors='black', s=50, zorder=10)
        texts.append(plt.text(x, y, feature, fontsize=9))
    
    # Auto-adjust text labels to avoid overlap
    adjust_text(texts, arrowprops=dict(arrowstyle='-', color='black'))
    
    plt.title(f"Mechanism of Action: {drug_code}\n(Volcano Plot)")
    plt.xlabel(f"Phenotypic Shift (Effect Size in {layer} units)")
    plt.ylabel("-Log10 P-Value")
    plt.axvline(0, linestyle='--', color='grey', alpha=0.5)
    plt.axhline(-np.log10(0.05), linestyle=':', color='blue', label='P=0.05')
    plt.tight_layout()
    plt.show()
    
    # --- PLOT 2: STRIP PLOTS OF TOP FEATURES ---
    # Visual validation of the top hits
    print(f"Top {top_n} differentiating features:")
    print(top_hits[['Effect_Size', 'P_Val']])
    
    # Melt data for Seaborn
    plot_df = pd.concat([drug_df[top_hits.index], ctrl_df[top_hits.index]])
    plot_df['Condition'] = [drug_code]*len(drug_df) + [control_code]*len(ctrl_df)
    melted = plot_df.melt(id_vars='Condition', var_name='Feature', value_name='Normalized Intensity')
    
    # 1. Setup the specific order
    # This ensures Handle 0 = DMSO Box, Handle 1 = Drug Box
    hue_order = [control_code, drug_code] 
    
    plt.figure(figsize=(15, 10))
    ax = plt.gca()
    
    # 2. Plot Boxplot (First Layer)
    # We use boxprops={'alpha': 0.4} to make boxes semi-transparent
    sns.boxplot(
        data=melted, 
        x='Feature', 
        y='Normalized Intensity', 
        hue='Condition',
        hue_order=hue_order,
        dodge=True, 
        boxprops={'alpha': 0.4}, # Lighter boxes so points stand out
        showfliers=False,        # Hide outliers (stripplot shows them)
        palette={drug_code: 'red', control_code: 'grey'},
        ax=ax
    )
    
    # 3. Plot Stripplot (Second Layer)
    # dodge=True aligns points with the boxes
    sns.stripplot(
        data=melted, 
        x='Feature', 
        y='Normalized Intensity', 
        hue='Condition',
        hue_order=hue_order,
        dodge=True, 
        jitter=True, 
        size=5, 
        color='black', # Keep points black or dark for contrast
        alpha=0.6,
        palette={drug_code: 'red', control_code: 'grey'},
        ax=ax
    )
    
    # 4. Create the Combined Legend
    # Get all handles (0,1 = Boxes; 2,3 = Points)
    handles, labels = ax.get_legend_handles_labels()
    
    # Combine them: (Box1, Point1), (Box2, Point2)
    # handles[0] matches hue_order[0] (DMSO Box)
    # handles[2] matches hue_order[0] (DMSO Points)
    combined_handles = [
        (handles[0], handles[2]), 
        (handles[1], handles[3])
    ]
    
    # Create the custom legend
    ax.legend(
        handles=combined_handles,
        labels=hue_order, # ['DMSO', 'Drug']
        loc='upper left', bbox_to_anchor=(1.05, 1), 
        handlelength=4, 
        handler_map={tuple: HandlerTuple(ndivide=None)}, # The magic instruction
        title='Condition'
    )
    
    plt.title(f"Mechanism of Action: {drug_code} vs {control_code}")
    plt.xlabel("Feature")
    plt.ylabel("Normalized Feature Intensity (Robust Z-Score)")
    plt.xticks(rotation=45, ha='right')
    plt.tight_layout()
    plt.show()
